1. Brute force:

Complexity: O(n2)O(n^2)O(n2)﻿

This is pretty straight forward. We iterate through the entire array and for each element we check the difference between the indices larger than the current index.

Code: 

int maxProfit(vector<int>& prices) {
	
	int maxProfit = 0;
	
	for(int i = 0; i < prices.size() - 1; i++) {
		for(int j = i + 1; j < prices.size(); j++) {
			maxProfit = max(maxProfit, prices[j] - prices[i])
		}
	}

	return maxProfit;
}

2. Array sum approach

Complexity: O(n)O(n)O(n)﻿

In this solution we calculate 2 arrays. The first which is calculated from left to right, denotes the minimum value in the array until this index from left. Code: 

int maxProfit(vector<int>& prices) {
        
    int size = prices.size();
    int maxProfit = 0, leftMin = 10001, rightMax = 0;
    int minArray[size], maxArray[size];
    
    
    for(int i = 0; i < size; i++) {
        
        if(prices[i] < leftMin) leftMin = prices[i];
        minArray[i] = leftMin;
        
        if(prices[size - 1 - i] > rightMax) rightMax = prices[size - 1 - i];
        maxArray[size - 1 - i] = rightMax;

    }
    
    for(int i = 0; i < size; i++)
        if(maxArray[i] - minArray[i] > maxProfit)
            maxProfit = maxArray[i] - minArray[i];
    
    return maxProfit;
}

int maxProfit(vector<int>& prices) {
        
    int maxProfit = 0, maxCurrent = 0;
    
    for(int i = 1; i < prices.size(); i++){
        maxCurrent = max(0, maxCurrent + prices[i] - prices[i - 1]);
        maxProfit = max(maxProfit, maxCurrent);
    }
    
    return maxProfit;
}
